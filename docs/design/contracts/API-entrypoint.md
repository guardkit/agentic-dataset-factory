## Startup Sequence

```python
def startup():
    """Startup sequence — fail fast on any validation error."""

    # 1. Load config
    config = load_config()

    # 2. Configure logging (structured JSON — ADR-ARCH-007)
    configure_logging(config.logging)

    # 3. Set LangSmith project
    os.environ["LANGSMITH_PROJECT"] = f"adf-{config.domain}"

    # 4. Resolve domain path
    domain_path = Path(f"domains/{config.domain}")
    if not domain_path.exists():
        raise DomainNotFoundError(config.domain)

    # 5. Parse and validate GOAL.md (strict — DDR-001)
    goal = parse_goal_md(domain_path / "GOAL.md")

    # 6. Check ChromaDB collection exists
    collection = get_chromadb_collection(config.domain)
    if collection.count() == 0:
        raise RuntimeError(
            f"No chunks found for '{config.domain}'. "
            f"Run: python -m ingestion.ingest --domain {config.domain}"
        )

    # 7. Clean output directory or resume (ADR-ARCH-008, ADR-ARCH-010)
    if resume and Path("output/.checkpoint").exists():
        start_index = read_checkpoint()
        log.info(f"Resuming from target {start_index}")
    else:
        clean_output_directory()
        start_index = 0

    # 8. Build prompts (base + GOAL.md injection)
    player_prompt = build_player_prompt(goal)
    coach_prompt = build_coach_prompt(goal)

    # 9. Create tools (bound to domain)
    tools = [
        create_rag_retrieval_tool(config.domain),
        create_write_output_tool(Path("output"), goal.metadata_schema),
    ]

    # 10. Instantiate agents via factories
    player = create_player(config.player, tools, player_prompt, ["./AGENTS.md"])
    coach = create_coach(config.coach, coach_prompt, ["./AGENTS.md"])

    # 11. Build generation targets from GOAL.md
    targets = build_generation_targets(goal)

    # 12. Run generation loop (sequential — ADR-ARCH-006)
    run_generation_loop(player, coach, targets, config.generation)
```

## Generation Loop

```python
def run_generation_loop(
    player: DeepAgent,
    coach: DeepAgent,
    targets: list[GenerationTarget],
    config: GenerationConfig,
) -> GenerationResult:
    """Run sequential Player-Coach generation loop.

    For each target:
      1. Player generates example (with RAG retrieval)
      2. Coach evaluates against criteria
      3. If accepted: Player writes via write_output
      4. If rejected and turns < max_turns: Player revises
      5. If rejected and turns >= max_turns: log to rejected.jsonl

    Returns:
        GenerationResult with statistics.
    """

@dataclass
class GenerationResult:
    total_targets: int
    accepted: int
    rejected: int
    total_turns: int
    elapsed_seconds: float
```
